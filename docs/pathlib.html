fn path_relative(source_ string, linkpath_ string) !string

recalc path between target & source we only support if source_ is an existing dir, links will not be supported a0 := pathlib.path_relative('$testpath/a/b/c', '$testpath/a/d.txt') or { panic(err) } assert a0 == '../../d.txt' a2 := pathlib.path_relative('$testpath/a/b/c', '$testpath/d.txt') or { panic(err) } assert a2 == '../../../d.txt' a8 := pathlib.path_relative('$testpath/a/b/c', '$testpath/a/b/c/d/e/e.txt') or { panic(err) } assert a8 == 'd/e/e.txt'
